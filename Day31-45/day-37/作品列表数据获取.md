* 接着处理下getWorks中的dispatch，reducers中的works及打印下index中的props 
    ```js
    import HTTP from './http'
    export default function getWorks(page){
        return function(dispatch){
            dispatch({
                type: "LOAD"
            })
            return HTTP.post(`/lecturer/lists?page=${page}&rows=8`, {
                order: "desc",
                sort: "sort",
                category_id: 1,
                recommend: 1
            })
            .then(res => {
            //    console.log(res);
            dispatch({
                type: "LOADOVER",
                data: res.data
            })
            })
        }
    }    
    ```  
    ```js
    export default function works(state={
        data: [],
        loading: false,
        loadEnd: false,
    }, action){
        switch(action.type){
            case "LOAD":
                return {
                    ...state,
                    loading: true
                };
            case "LOADOVER":
                return {
                    ...state,
                    loading: false,
                    data: state.data.concat(action.data), //这里拼接下数据
                }  
            case "LOADEND":
                return {
                    ...state,
                    loadEnd: true
                }      
        }
        return state;
    }    
    ```
