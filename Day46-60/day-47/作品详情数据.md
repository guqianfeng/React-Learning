# 作品详情数据

## 练习

* 在reducer下新建work.js
```js
export default function work(state = {
  data: {},
  loading: false,
}, action) {
  switch (action.type) {
    case "WORK_RESET":
      return {
        data: {},
        loading: false
      };
    case "WORK_LOADOVER":
      return {
        loading: true,
        data: action.data
      }
  }
  return state;
}
```

* action新建getWork.js
```js
import HTTP from './http'
export default function getWork(id) {
  return function (dispatch) {
    return HTTP.post(`/lecturer/info`, {
      article_id: id
    })
      .then(res => {
        //  console.log(res);
        dispatch({
          type: 'WORK_LOADOVER',
          data: res.data
        })
      })
  }
}
```

* index.js
```js
import React, { useEffect } from 'react'
import { connect } from 'react-redux'
import Frame from '../../common/component/frame'

import '../../common/css/miiaov.css'

import getWork from '../../store/action/getWork'

function Work (props) {
  let { data, loading, dispatch, match } = props
  let { id } = match.params
  console.log(data, loading)
  useEffect(() => {
    dispatch(getWork(id))
    return () => {
      dispatch({
        type: 'WORK_RESET'
      })
    }
  }, [])
  return (
    <div>
      <Frame />
      <footer className="miiapv_footer">
        回复本帖
      </footer>
    </div>
  )
}

export default connect(state => ({...(state.work)}))(Work)
```
